N <- 300
x1 <- rnorm(N, 5, 5)
x2 <- rnorm(N, 2, 1)
x3 <- runif(N, 2, 4)

e1 <- rnorm(N, 0, 1)

# good data to fit
y <- x1 + 3*x2 + 0.1 * x3 + e1

model1 <- lm(y ~ x1 + x2 + x3)
op <- par(mfrow=c(2,2))
plot(model1)


par(op)


# let's add some outliers
y.out <- y
y.out[5] <- 1000
y.out[21] <- 2412

model2 <- lm(y.out ~ x1 + x2 + x3)
op <- par(mfrow=c(2,2))
plot(model2)


par(op)

# new strange errors
y.strange <- y + rnorm(N, 0, 10)^5

model3 <- lm(y.strange ~ x1 + x2 + x3)
op <- par(mfrow=c(2,2))
plot(model3)


par(op)
